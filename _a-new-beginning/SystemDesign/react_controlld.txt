function UncontrolledInput() {
  const inputRef = useRef();

  const handleSubmit = () => {
    alert(inputRef.current.value);
  };

  return (
    <div>
      <input ref={inputRef} defaultValue="Hello" />
      <button onClick={handleSubmit}>Show Value</button>
    </div>
  );
}


No React state involved

You only read the value when you need it (like form submission)

Useful for non-critical inputs, file uploads, or performance optimizations
